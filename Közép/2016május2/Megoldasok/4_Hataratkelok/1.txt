FROM hatar
WHERE tipus='vasúti'

FROM hatar AS h1, hatar AS h2
WHERE h1.magyarh=h2.magyarh and h1.tipus='közúti' and h2.tipus='vasúti';

FROM hatar AS h1, hatar AS h2
WHERE h1.magyarh=h2.magyarh and h1.tipus='közúti' and h2.tipus='vasúti'
